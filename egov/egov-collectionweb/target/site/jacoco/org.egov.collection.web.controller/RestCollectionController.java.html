package org.egov.collection.web.controller;

import org.egov.collection.integration.models.BillAccountDetails;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.HashMap;
import java.util.Map;

@RestController
@RequestMapping(value = { "/public" })
public class RestCollectionController {
    @GetMapping(value = "/getPurpose")
    public Map<BillAccountDetails.PURPOSE, BillAccountDetails.PURPOSE> getPurpose() {
        Map<BillAccountDetails.PURPOSE, BillAccountDetails.PURPOSE> purpose = new HashMap<>();
        for (BillAccountDetails.PURPOSE key : BillAccountDetails.PURPOSE.values()) {
            purpose.put(key, BillAccountDetails.PURPOSE.valueOf(key.toString()));
        }
        return purpose;
    }
}
