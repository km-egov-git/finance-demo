	@Override
	public int vote(Authentication authentication, Object object, Collection attributes) {
		
	    if(null==authentication || ! (authentication.getPrincipal() instanceof CurrentUser))
	        return ACCESS_DENIED;
			
	return ACCESS_GRANTED;
	}
